```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {

	ofstream outFile("example.txt"); // 创建或打开文件
    
    if (!outFile) {
        cerr << "无法打开文件！" << endl;
        return 1;
    }
    
    outFile << "这是第一行文本\n";
    outFile << "这是第二行文本\n";
    outFile << 12345 << "\n";
    
    outFile.close(); // 关闭文件

	
	ifstream inFile("example.txt");

    if (!inFile) {
        cerr << "无法打开文件！" << endl;
        return 1;
    }
    
    string line;
    while (getline(inFile, line)) {
        cout << line << endl;
    }
    
    inFile.close();

	/*
	std::ios::in：打开用于读取

	std::ios::out：打开用于写入

	std::ios::app：追加到文件末尾

	std::ios::ate：打开后定位到文件末尾

	std::ios::trunc：如果文件存在则截断

	std::ios::binary：二进制模式
	std::fstream file("data.dat", std::ios::in | std::ios::out | std::ios::binary);
	*/
    return 0;
}
```
