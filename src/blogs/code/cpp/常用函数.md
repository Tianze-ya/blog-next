#include <bits/stdc++.h>

int main() {

	fill(起始地址,结束地址,填充物) 填充

	time(0) - 获取当前时间
	tolower(char) / toupper() - 大小写转换

	三目运算符
	A ？B : C;
	A均为bool
	B,C为值
